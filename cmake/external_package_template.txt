# external package CMakeLists.txt template

# minimum cmake version requirement
cmake_minimum_required(VERSION 3.5.1)

# project name
project(name)

# find package
find_package(PACKAGE_NAME REQUIRED)

# add header-only target
add_library(${PROJECT_NAME} INTERFACE)

# add compile flags
# if no flags need to be set, this command needs to be commented
target_compile_options(${PROJECT_NAME}
	INTERFACE ${PACKAGE_NAME_CXX_COMPILE_FLAGS}
)

# add include directories
target_include_directories(${PROJECT_NAME}
	INTERFACE ${PACKAGE_NAME_CXX_INCLUDE_PATH}
)

# add library dependencies
target_link_libraries(${PROJECT_NAME}
	INTERFACE ${PACKAGE_NAME_CXX_LINK_FLAGS}
	INTERFACE ${PACKAGE_NAME_CXX_LIBRARIES}
)
