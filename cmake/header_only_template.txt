# header-only CMakeLists.txt template

# minimum cmake version requirement
cmake_minimum_required(VERSION 3.5.1)

# project name
project(name)

# add header-only target
add_library(${PROJECT_NAME} INTERFACE)

# add compile flags
# if no flags need to be set, this command needs to be commented
target_compile_options(${PROJECT_NAME}
	INTERFACE ${CXX_COMPILE_FLAGS}
)

# add include directories
target_include_directories(${PROJECT_NAME}
	INTERFACE ${CMAKE_CURRENT_SOURCE_DIR}/include
)

# add library dependencies
target_link_libraries(${PROJECT_NAME}
	INTERFACE other_project
	INTERFACE path/to/lib/libmylib.a
	INTERFACE ${CXX_LINK_FLAGS}
)
